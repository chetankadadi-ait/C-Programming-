15)evenandoddcount
#include <stdio.h>
#include <stdlib.h>

void countevenodd(int arr[], int size)
{
    int even = 0, odd = 0;

    printf("Even numbers: ");
    for (int i = 0; i < size; i++)
    {
        if (arr[i] % 2 == 0)
        {
            printf("%d ", arr[i]);
            even++;
        }
    }

    printf("\nOdd numbers: ");
    for (int i = 0; i < size; i++)
    {
        if (arr[i] % 2 != 0)
        {
            printf("%d ", arr[i]);
            odd++;
        }
    }

    printf("\nEven count: %d", even);
    printf("\nOdd count: %d\n", odd);
}

{
    int arr[8];
