#include <stdio.h>

int main(void) {
    int x = 10;
    int *p = &x;   // p holds the address of x

    printf("Value of x     : %d\n", x);
    printf("Address of x   : %p\n", (void * )&x);
    printf("Value of p     : %p\n", (void * )p);    // same as &x
    printf("Value via *p   : %d\n", *p);           // same as x

    *p = 50;   // modify x through the pointer
    printf("x after *p=50  : %d\n", x);            // x is now 50

    return 0;
}

	• Declares an integer variable num = 25
	• Declares a pointer ptr and assigns it the address of num
	• Prints the value of num, the address of num, the value stored in ptr, and the value pointed to by ptr
	• Verifies that &num and ptr print the same address

Expected output (addresses will vary):

Value of num    : 25
Address of num  : 0x7ffc...
Value of ptr    : 0x7ffc...
Value via *ptr  : 25
#include <stdio.h>
#include <stdlib.h>

int main()
{
   int num=25;
   int *ptr=&num;
   printf("enter the value of num %d\n",num);
   printf("enter the adress of num %p\n",(void *)&num);
   printf("enter the adress of *ptr %p\n",(void*) ptr);
   printf("value of *ptr %d\n",*ptr);
}
